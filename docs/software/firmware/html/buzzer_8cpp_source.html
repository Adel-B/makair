101 void Buzzer_Init() {
102  // Automatically retrieve timer instance and channel associated with the pin
103  // Useful in case of board change
104  TIM_TypeDef* Instance = reinterpret_cast<TIM_TypeDef*>(
105  pinmap_peripheral(digitalPinToPinName(PIN_BUZZER), PinMap_PWM));
106  BuzzerTimerChannel =
107  STM_PIN_CHANNEL(pinmap_function(digitalPinToPinName(PIN_BUZZER), PinMap_PWM));
108 
109  // Buzzer HardwareTimer object creation
110  BuzzerTim = new HardwareTimer(Instance);
111 
112  BuzzerTim->setMode(BuzzerTimerChannel, TIMER_OUTPUT_COMPARE_FORCED_INACTIVE, PIN_BUZZER);
113  // 4 ticks = 1 ms (it is not possible to have 1 tick = 1 ms because prescaler is 16 bit and
114  // input frequency either 84 or 100 MHz Use of tick format to avoid computation within interrupt
115  // handler
116  BuzzerTim->setPrescaleFactor(BuzzerTim->getTimerClkFreq() / (TIMER_TICK_PER_MS * 1000));
117  BuzzerTim->setOverflow(100 * TIMER_TICK_PER_MS, TICK_FORMAT); // Default 100milisecondes
118 
119  // Start with inactive state without interruptions
120  BuzzerTim->resume();
121 }
